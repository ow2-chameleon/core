Maven

The chameleon-core released artifacts are available on Maven Central:

<!-- The distribution -->
<dependency>
    <groupId>org.ow2.chameleon</groupId>
    <artifactId>chameleon-core</artifactId>
    <version>1.0.2</version>
    <type>zip</type>
</dependency>
<!-- The main jar file -->
<dependency>
    <groupId>org.ow2.chameleon</groupId>
    <artifactId>chameleon-core</artifactId>
    <version>1.0.2</version>
</dependency>

Snapshots artifacts are available on the ow2.snapshots repository:

<repositories>
    <repository>
        <id>ow2.snapshots</id>
        <url>http://repository.ow2.org/nexus/content/repositories/snapshots/</url>
        <snapshots>
            <enabled>true</enabled>
        </snapshots>
        <releases>
            <enabled>false</enabled>
        </releases>
    </repository>
</repositories>
